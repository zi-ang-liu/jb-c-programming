## Solutions
``` c title="9-1.c"
#include <stdio.h>
double average(int num1, int num2);

int main(void){
	int x,y;
	double ave;
	scanf("%d",&x);
	scanf("%d",&y);
	
	ave = average(x,y);
	printf("average of %d and %d = %f\n",x,y,ave);
	
	return 0;
}

double average(int num1, int num2){
	return (num1 + num2)/2.0;
}
```

``` c title="9-2.c"
#include <stdio.h>
double average2(int s[], int len);

int main(void){
	int a[] = {2,4,6,1,5};
	printf("%f\n", average2(a,5));
	return 0;
}

double average2(int s[], int len){
	double sum=0;
	int i;
	for(i=0; i<len; i++){
		sum += s[i];
	}
	return sum/len;
}
```

``` c title="9-3.c"
#include <stdio.h>
void nibai(int s[], int len){
	int i;
	for(i=0;i<len;i++){
		s[i] = s[i] * 2;
	}
```

``` c title="9-4.c"
#include <stdio.h>

void replace(char s[])
{
	int i=0;
	while(s[i]){
		if(s[i]=='a') s[i]='@';
		i++;
	}
}

int main(void)
{
	char st[]="Okadai Taro";
	puts(st);
	replace(st);
	puts(st);
	return 0;
}
```
